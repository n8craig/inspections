                 sheet="Inspections",
                 col_names = TRUE,
                 col_types = "c") %>% 
  clean_names() %>% 

  # Removing a curious whitespace character that appears
  # in more recent facility names
  mutate(facility = str_replace_all(facility,
         pattern = " ",
         replacement = " ")) %>% 
  
  separate(.,
           col = facility,
           into = c("facility","inspection_date"),
           sep = "\\) - ") %>%
  separate(.,
           col = facility,
           into = c("facility","state"),
           sep = "\\(")
